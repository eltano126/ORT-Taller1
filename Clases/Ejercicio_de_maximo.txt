#include <iostream>
using namespace std;

void  main ()
{
// Declaración de variables
	int num; // Guarda el numero ingresado
	int max; // Guarda el valor máximo

// Desarrollo del programa.
	
// Pedimos un numero para el programa
	cout << "Por favor ingrese un numero.0 (cero) Para terminar." << endl;
	cout << "Presione ENTER para continuar";
	cin >> num;
	getchar ();

// Guardamos el numero como máximo
	max = num;


	while (num != 0)
	{
		if (num > max)
		{
			max = num;
		}
		cout << "Por favor ingrese un numero.0 (cero) Para terminar." << endl;
	cout << "Presione ENTER para continuar" ;
	cin >> num;
	getchar ();
	}
		if (max != 0)
		{
			cout << "El máximo es " << max;
			getchar ();
		}
		else
		{
			cout << "No hay ningun maximo";
		}

}
